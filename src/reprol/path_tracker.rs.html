1//! PathTracker
2//!
3//! 単一始点最短路探索アルゴリズムにおいて， 経路情報を管理するためのモジュール．
4//!
5//! `PathTracker`トレイトとその2個の実装を提供する:
6//! - `WithPath`: 経路を保存し，始点から終点までの経路を構築できる．
7//! - `NoPath`: 経路情報を保存しない場合のダミー．
8//!
9//! bfsやdijkstraなどのモジュールで使用する．
10
11/// 経路情報を管理するためのトレイト．
12pub trait PathTracker<V> {
13    fn new(n: usize) -> Self;
14
15    /// `index`の直前の頂点を返す．
16    fn get_previous(&self, index: usize) -> Option<&V>;
17
18    /// `index`の直前の頂点を`v`に更新する．
19    fn set_previous(&mut self, index: usize, v: &V);
20
21    /// 始点から`end`までの経路を構築する．
22    /// 到達判定は行わない．
23    fn construct_path(&self, to_index: &impl Fn(&V) -> usize, end: &V) -> Vec<V>;
24}
25
26/// 経路を保存する場合に用いる構造体．
27/// 各頂点の直前の頂点を保存する．
28pub struct WithPath<V> {
29    previous: Vec<Option<V>>,
30}
31
32impl<V: Clone> PathTracker<V> for WithPath<V> {
33    fn new(n: usize) -> Self {
34        Self {
35            previous: vec![None; n],
36        }
37    }
38
39    fn get_previous(&self, index: usize) -> Option<&V> {
40        self.previous[index].as_ref()
41    }
42
43    fn set_previous(&mut self, index: usize, value: &V) {
44        self.previous[index] = Some(value.clone());
45    }
46
47    fn construct_path(&self, to_index: &impl Fn(&V) -> usize, end: &V) -> Vec<V> {
48        let mut v = end;
49        let mut path = vec![v];
50
51        while let Some(pv) = self.previous[to_index(v)].as_ref() {
52            path.push(pv);
53            v = pv;
54        }
55
56        path.into_iter().rev().cloned().collect()
57    }
58}
59
60/// 経路を保存しない場合に用いる構造体(ダミー)．
61pub struct NoPath;
62
63impl<V> PathTracker<V> for NoPath {
64    #[inline(always)]
65    fn new(_: usize) -> Self {
66        Self
67    }
68
69    #[inline(always)]
70    fn get_previous(&self, _: usize) -> Option<&V> {
71        None
72    }
73
74    #[inline(always)]
75    fn set_previous(&mut self, _: usize, _: &V) {}
76
77    #[inline(always)]
78    fn construct_path(&self, _: &impl Fn(&V) -> usize, _: &V) -> Vec<V> {
79        panic!("NoPath::construct_path should not be called.");
80    }
81}
