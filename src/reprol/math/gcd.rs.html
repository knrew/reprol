5macro_rules! impl_signed {
6    ($($ty:ident),*) => {$(
7        impl Gcd for $ty {
8            fn gcd(self, rhs: Self) -> Self {
9                if rhs == 0 {
10                    self.abs()
11                } else {
12                    rhs.gcd(self % rhs)
13                }
14            }
15        }
16    )*};
17}
18
19impl_signed! { i8, i16, i32, i64, i128, isize }
