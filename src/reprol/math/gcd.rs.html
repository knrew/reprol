1//! 最大公約数(greatest common divisor)
2//!
3//! 最大公約数(gcd)を計算する．
4//!
5//! # 使用例
6//! ```
7//! use reprol::math::gcd::Gcd;
8//! assert_eq!(48u64.gcd(18), 6);
9//! ```
10
11pub trait Gcd {
12    fn gcd(self, rhs: Self) -> Self;
13}
14
15macro_rules! impl_gcd_unsigned {
16    ($ty: ty) => {
17        impl Gcd for $ty {
18            fn gcd(self, rhs: Self) -> Self {
19                let (mut a, mut b) = (self, rhs);
20                while b != 0 {
21                    (a, b) = (b, a % b)
22                }
23                a
24            }
25        }
26    };
27}
28
29macro_rules! impl_gcd_signed {
30    ($ty: ty) => {
31        impl Gcd for $ty {
32            fn gcd(self, rhs: Self) -> Self {
33                let (mut a, mut b) = (self, rhs);
34                while b != 0 {
35                    (a, b) = (b, a.rem_euclid(b));
36                }
37                a.abs()
38            }
39        }
40    };
41}
42
43macro_rules! impl_gcd_for {
44    (unsigned: [$($u:ty),* $(,)?], signed: [$($s:ty),* $(,)?]$(,)?) => {
45        $( impl_gcd_unsigned!($u); )*
46        $( impl_gcd_signed!($s); )*
47    };
48}
49
50impl_gcd_for! {
51    unsigned: [u8, u16, u32, u64, u128, usize],
52    signed:   [i8, i16, i32, i64, i128, isize],
53}
