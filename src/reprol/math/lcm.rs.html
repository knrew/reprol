1use crate::math::gcd::Gcd;
2
3pub trait Lcm: Gcd {
4    fn lcm(self, rhs: Self) -> Self;
5}
6
7macro_rules! impl_signed {
8    ($($ty:ident),*) => {$(
9        impl Lcm for $ty {
10            fn lcm(self, rhs: Self) -> Self {
11                let m = self.abs();
12                let n = rhs.abs();
13                if m == 0 || n == 0 {
14                    return 0;
15                } else {
16                    m / self.gcd(rhs) * n
17                }
18            }
19        }
20    )*};
21}
22
23impl_signed! { i8, i16, i32, i64, i128, isize }
24
25macro_rules! impl_unsigned {
26    ($($ty:ident),*) => {$(
27        impl Lcm for $ty {
28            fn lcm(self, rhs: Self) -> Self {
29                if self == 0 || rhs == 0 {
30                    0
31                } else {
32                    self / self.gcd(rhs) * rhs
33                }
34            }
35        }
36
37    )*};
38}
39
40impl_unsigned! { u8, u16, u32, u64, u128, usize }
