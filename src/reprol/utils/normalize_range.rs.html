1//! 区間正規化ユーティリティ
2//!
3//! 任意の `RangeBounds` を半開区間 `Range<T>` に正規化する．
4//! Rust の範囲構文(`..`, `a..b`, `a..=b`, `..b`, `..=b`, `a..`)を
5//! 統一的に扱うための変換レイヤ．
6//!
7//! # Examples
8//!
9//! ```ignore
10//! use reprol::utils::normalize_range::{normalize, normalize_index};
11//!
12//! // 閉区間 [2, 5] → 半開区間 [2, 6)
13//! assert_eq!(normalize(2..=5, 0, 10), 2..6);
14//!
15//! // 非有界 .. → [0, 10)
16//! assert_eq!(normalize(.., 0, 10), 0..10);
17//!
18//! // インデックス区間の正規化(配列長 = 5)
19//! assert_eq!(normalize_index(1..=3, 5), 1..4);
20//! assert_eq!(normalize_index(.., 5), 0..5);
21//! ```
22
23use std::ops::{Bound, Range, RangeBounds};
24
25/// 離散型の後続値を定義するトレイト．
26///
27/// `Bound::Included(x)` を `Bound::Excluded(x + 1)` に変換するために使用する．
28///
29/// # Notes
30///
31/// 整数型の最大値(`i64::MAX`, `usize::MAX` など)に対して `successor` を呼ぶと
32/// オーバーフローする．閉区間の上端に最大値を使う場合は注意が必要．
33pub trait Discrete: Sized + Clone + PartialOrd {
34    /// 後続の値を返す．
35    fn successor(&self) -> Self;
36}
37
38macro_rules! impl_discrete_inner {
39    ($ty:ty) => {
40        impl Discrete for $ty {
41            #[inline]
42            fn successor(&self) -> Self {
43                self + 1
44            }
45        }
46    };
47}
48
49macro_rules! impl_discrete {
50    ($($ty:ty),* $(,)?) => {
51        $( impl_discrete_inner!($ty); )*
52    };
53}
54
55impl_discrete! {
56    i8, i16, i32, i64, i128, isize,
57    u8, u16, u32, u64, u128, usize,
58}
59
60/// 任意の `RangeBounds` を半開区間 `[min, sup)` に正規化する．
61///
62/// 非有界(`Unbounded`)の開始は `min`，終了は `sup` に置き換える．
63///
64/// # Panics
65///
66/// 正規化後の区間が `[min, sup)` の範囲外になる場合，または逆転している場合にパニックする．
67pub fn normalize<T: Discrete>(range: impl RangeBounds<T>, min: T, sup: T) -> Range<T> {
68    let l = match range.start_bound() {
69        Bound::Unbounded => min.clone(),
70        Bound::Included(x) => x.clone(),
71        Bound::Excluded(x) => x.successor(),
72    };
73
74    let r = match range.end_bound() {
75        Bound::Unbounded => sup.clone(),
76        Bound::Excluded(x) => x.clone(),
77        Bound::Included(x) => x.successor(),
78    };
79
80    assert!(min <= l && l <= r && r <= sup);
81
82    l..r
83}
84
85/// インデックス区間を半開区間 `[0, len)` に正規化する．
86///
87/// `normalize(range, 0, len)` の省略形．
88#[inline]
89pub fn normalize_index(range: impl RangeBounds<usize>, len: usize) -> Range<usize> {
90    normalize(range, 0, len)
91}
