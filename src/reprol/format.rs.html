3use std::fmt::{Display, Write};
4
5pub trait IterFormatter {
6    /// 要素のイテレータからsep区切りで文字列に変換する．
7    fn to_string(&mut self, sep: &str) -> String;
8}
9
10impl<I> IterFormatter for I
11where
12    I: Iterator,
13    I::Item: Display,
14{
15    fn to_string(&mut self, sep: &str) -> String {
16        let mut res = String::new();
17        if let Some(item) = self.next() {
18            write!(&mut res, "{}", item).unwrap();
19            while let Some(item) = self.next() {
20                write!(&mut res, "{}{}", sep, item).unwrap();
21            }
22        }
23        res
24    }
25}

27pub trait ArrayFormatter {
28    /// 配列をsep区切りで文字列に変換する．
29    fn to_string(&self, sep: &str) -> String;
30}
31
32impl<T> ArrayFormatter for [T]
33where
34    T: Display,
35{
36    fn to_string(&self, sep: &str) -> String {
37        self.iter().to_string(sep)
38    }
39}

41pub trait Usize1ArrayFormatter {
42    /// usizeの配列を1-indexedの文字列に変換する．
43    fn to_string_usize1(&self, sep: &str) -> String;
44}
45
46impl Usize1ArrayFormatter for [usize] {
47    fn to_string_usize1(&self, sep: &str) -> String {
48        self.iter().map(|i| i + 1).to_string(sep)
49    }
50}

52pub trait CharsFormatter {
53    // char型またはu8型の配列を文字列に変換する．
54    fn as_string(&self) -> String;
55}
56
57impl CharsFormatter for [char] {
58    fn as_string(&self) -> String {
59        self.iter().collect::<String>()
60    }
61}
62
63impl CharsFormatter for [u8] {
64    fn as_string(&self) -> String {
65        self.iter().map(|&c| c as char).collect::<String>()
66    }
67}
