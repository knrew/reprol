1use std::mem::swap;
2
3use crate::ops::monoid::Monoid;
4
5pub struct MonoidDsu<O: Monoid> {
6    parents: Vec<usize>,
7    sizes: Vec<usize>,
8    states: Vec<O::Value>,
9    count_components: usize,
10    monoid: O,
11}
12
13impl<O: Monoid> MonoidDsu<O> {
14    pub fn new(initial_states: Vec<O::Value>, monoid: O) -> Self {
15        let n = initial_states.len();
16        Self {
17            parents: (0..n).collect(),
18            sizes: vec![1; n],
19            states: initial_states,
20            count_components: n,
21            monoid,
22        }
23    }
24
25    /// xのrootのindexを返す
26    pub fn find(&mut self, v: usize) -> usize {
27        debug_assert!(v < self.parents.len());
28
29        if self.parents[v] == v {
30            return v;
31        }
32        let root = self.find(self.parents[v]);
33
34        self.parents[v] = root;
35        root
36    }
37
38    /// xが属するグループとyが属するグループを統合する
39    pub fn merge(&mut self, u: usize, v: usize) {
40        let mut u = self.find(u);
41        let mut v = self.find(v);
42
43        if u == v {
44            return;
45        }
46
47        self.count_components -= 1;
48
49        if self.sizes[u] < self.sizes[v] {
50            swap(&mut u, &mut v);
51        }
52
53        self.parents[v] = u;
54        self.sizes[u] += self.sizes[v];
55        self.states[u] = self.monoid.op(&self.states[u], &self.states[v]);
56    }
57
58    /// xとyが同じグループに属すか
59    pub fn connected(&mut self, u: usize, v: usize) -> bool {
60        self.find(u) == self.find(v)
61    }
62
63    /// xが属するグループの要素数
64    pub fn size(&mut self, v: usize) -> usize {
65        let v = self.find(v);
66        self.sizes[v]
67    }
68
69    pub fn state(&mut self, v: usize) -> &O::Value {
70        let v = self.find(v);
71        &self.states[v]
72    }
73
74    /// 連結成分の個数
75    pub fn count_components(&self) -> usize {
76        self.count_components
77    }
78}
