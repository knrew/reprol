1//! Fenwick Tree(Binary Indexed Tree)
2//!
3//! (群をなす集合の要素からなる)配列を管理するデータ構造．
4//! 要素の1点更新と区間積の取得をO(logN)で行うことができる．
5//!
6//! # 使用例
7//! ```
8//! use reprol::{ds::fenwick_tree::FenwickTree, ops::op_add::OpAdd};
9//! let mut ft = FenwickTree::<OpAdd<i32>>::new(5);
10//! ft.op(1, &5); // v[1] += 5
11//! ft.op(2, &3); // v[2] += 3
12//! ft.op(4, &2); // v[4] += 2
13//! assert_eq!(ft.fold(..2), 5); // 区間[0, 2)の区間和
14//! assert_eq!(ft.fold(..=2), 8); // 区間[0, 2]の区間和
15//! assert_eq!(ft.fold(2..5), 5); // 区間[2, 5)の区間和
16//! ```
17

25/// Fenwick Tree
26pub struct FenwickTree<O: Group> {
27    nodes: Vec<O::Value>,
28    op: O,
29}
30
31impl<O: Group> FenwickTree<O> {
32    /// 長さ`n`で初期化する．
33    /// 要素はすべて単位元で初期化される．
34    pub fn new(n: usize) -> Self
35    where
36        O: Default,
37    {
38        Self::with_op(n, O::default())
39    }
40
41    /// 演算`op`を指定して長さ`n`で初期化する．
42    pub fn with_op(n: usize, op: O) -> Self {
43        Self {
44            nodes: (0..n).map(|_| op.identity()).collect(),
45            op,
46        }
47    }
48
49    /// `index`番目の要素に`rhs`を作用させる．
50    /// `v[i] <- v[i] * rhs`
51    pub fn op(&mut self, mut index: usize, rhs: &O::Value) {
52        assert!(index < self.nodes.len());
53        index += 1;
54        while index <= self.nodes.len() {
55            self.nodes[index - 1] = self.op.op(&self.nodes[index - 1], rhs);
56            index += index & index.wrapping_neg();
57        }
58    }
59
60    /// `index`番目の要素の値を`value`にする．
61    pub fn set(&mut self, index: usize, value: O::Value) {
62        let diff = self.op.op(&value, &self.op.inv(&self.get(index)));
63        self.op(index, &diff);
64    }
65
66    /// `index`番目の要素の値を返す．
67    pub fn get(&self, index: usize) -> O::Value {
68        self.fold(index..=index)
69    }
70
71    /// 区間`[0, r)`の区間積を返す．
72    fn cum(&self, mut r: usize) -> O::Value {
73        assert!(r <= self.nodes.len());
74        let mut res = self.op.identity();
75        while r > 0 {
76            res = self.op.op(&res, &self.nodes[r - 1]);
77            r -= r & r.wrapping_neg();
78        }
79        res
80    }
81
82    /// 区間`[l, r)`の区間積を返す．
83    pub fn fold(&self, range: impl RangeBounds<usize>) -> O::Value {
84        let Range { start: l, end: r } = to_open_range(range, self.nodes.len());
85        assert!(l <= r);
86        let cl = self.cum(l);
87        let cr = self.cum(r);
88        self.op.op(&cr, &self.op.inv(&cl))
89    }
90}

92impl<O: Group> From<(Vec<O::Value>, O)> for FenwickTree<O> {
93    fn from((v, op): (Vec<O::Value>, O)) -> Self {
94        let mut res = Self::with_op(v.len(), op);
95        v.into_iter()
96            .enumerate()
97            .for_each(|(i, rhs)| res.op(i, &rhs));
98        res
99    }
100}

102impl<O: Group, const N: usize> From<([O::Value; N], O)> for FenwickTree<O> {
103    fn from((v, op): ([O::Value; N], O)) -> Self {
104        Self::from((v.into_iter().collect::<Vec<_>>(), op))
105    }
106}

148#[cfg(test)]
149mod tests {
150    use rand::{Rng, SeedableRng, rngs::StdRng};

152    use crate::ops::op_add::OpAdd;

156    #[test]
157    fn test_sum() {
158        let mut ft = FenwickTree::<OpAdd<i64>>::new(10);
159        ft.op(0, &5);
160        ft.op(2, &10);
161        ft.op(6, &20);
162        assert_eq!(ft.fold(..1), 5);
163        assert_eq!(ft.fold(..3), 15);
164        assert_eq!(ft.fold(..7), 35);
165        assert_eq!(ft.fold(..), 35);
166        assert_eq!(ft.fold(0..3), 15);
167        assert_eq!(ft.fold(3..=6), 20);
168        ft.op(9, &10);
169        assert_eq!(ft.fold(0..10), 45);
170    }
171
172    #[test]
173    fn test_sum_random() {
174        macro_rules! define_test_function {
175            ($name:ident, $ty:ident) => {
176                fn $name(rng: &mut StdRng, mn: $ty, mx: $ty) {
177                    const T: usize = 100;
178                    const N: usize = 100;
179                    const Q: usize = 1000;
180
181                    for _ in 0..T {
182                        let mut ft = FenwickTree::<OpAdd<_>>::new(N);
183                        let mut naive = vec![0; N];
184
185                        for _ in 0..Q {
186                            // add
187                            // v[index] += d
188                            let index = rng.random_range(0..N);
189                            let d = rng.random_range(mn..=mx);
190                            ft.op(index, &d);
191                            naive[index] += d;
192
193                            // [l, r)の区間和を求める
194                            let l = rng.random_range(0..N);
195                            let r = rng.random_range(l..=N);
196                            assert_eq!(ft.fold(l..r), naive[l..r].iter().sum());
197                        }
198
199                        for l in 0..N {
200                            for r in l..=N {
201                                assert_eq!(ft.fold(l..r), naive[l..r].iter().sum());
202                            }
203                        }
204                    }
205                }
206            };
207        }
208
209        define_test_function!(test_i64, i64);
210        define_test_function!(test_u64, u64);
211
212        let mut rng = StdRng::seed_from_u64(30);
213
214        test_i64(&mut rng, -1000000000, 1000000000);
215        test_u64(&mut rng, 0, 1000000000);
216    }
