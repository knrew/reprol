1//! 累積積(累積和)
2//!
3//! モノイドの配列に対する累積積を管理するデータ構造．
4//! 群に対しては区間積[`fold`](CumulativeArray::fold)も提供する．
5//!
6//! # 使用例
7//! ## 累積和
8//! ```
9//! use reprol::ds::cumulative_array::CumulativeSum;
10//! let v = vec![1, 2, 3, 4, 5];
11//! let cum = CumulativeSum::new(v);
12//! assert_eq!(cum[3], 6); // [0, 3)の区間和
13//! assert_eq!(cum.fold(1..4), 9); // [1, 4)の区間和
14//! ```
15
16//! ## 累積最小値
17//! minは群ではないので`fold`は使えない．
18//! ```
19//! use reprol::{ds::cumulative_array::CumulativeArray, ops::op_min::OpMin};
20//! let v = vec![3, 5, 4, 1, 5];
21//! let cum = CumulativeArray::<OpMin<i32>>::new(v);
22//! assert_eq!(cum[3], 3); // [0, 3)の最小値
23//! assert_eq!(cum[4], 1); // [0, 4)の最小値
24//! ```
25//!
26//! ## 演算を定義する
27//! 演算(モノイドや群)を自分で定義して区間積を計算できる．
28//! 排他的論理和(xor)を定義した場合の例を示す．
29//! ```
30//! use reprol::{
31//!    ds::cumulative_array::CumulativeArray,
32//!    ops::{group::Group, monoid::Monoid},
33//! };
34//! #[derive(Default)]
35//! struct Op;
36//! impl Monoid for Op {
37//!    type Value = u32;
38//!    // 単位元
39//!    fn identity(&self) -> Self::Value {
40//!        0
41//!    }
42//!    // 演算
43//!    fn op(&self, x: &Self::Value, y: &Self::Value) -> Self::Value {
44//!        x ^ y
45//!    }
46//! }
47//! impl Group for Op {
48//!     // 逆元
49//!    fn inv(&self, &x: &Self::Value) -> Self::Value {
50//!        x
51//!    }
52//! }
53//! let v = vec![3, 1, 4, 1, 5, 9, 2];
54//! let cum = CumulativeArray::<Op>::new(v);
55//! assert_eq!(cum[3], 6); // [0, 3)の区間xor
56//! assert_eq!(cum.fold(1..5), 1); // [1, 5)の区間xor
57//!```

70/// 累積積を管理するデータ構造
71pub struct CumulativeArray<O: Monoid> {
72    data: Vec<O::Value>,
73    op: O,
74}
75
76impl<O: Monoid> CumulativeArray<O> {
77    /// 配列の累積配列を構築する．
78    pub fn new(v: Vec<O::Value>) -> Self
79    where
80        O: Default,
81    {
82        Self::with_op(v, O::default())
83    }
84
85    /// `i`番目の要素が`f(i)`であるような累積配列を構築する．
86    pub fn construct(len: usize, f: impl FnMut(usize) -> O::Value) -> Self
87    where
88        O: Default,
89    {
90        Self::construct_with_op(len, O::default(), f)
91    }
92
93    /// 演算`op`を明示的に渡して配列の累積配列を構築する．
94    pub fn with_op(v: Vec<O::Value>, op: O) -> Self {
95        assert!(!v.is_empty());
96        let mut data = Vec::with_capacity(v.len() + 1);
97        data.push(op.identity());
98        for i in 0..v.len() {
99            data.push(op.op(&data[i], &v[i]));
100        }
101        Self { data, op }
102    }
103
104    /// 演算`op`を明示的に渡して`i`番目の要素が`f(i)`であるような累積配列を構築する．
105    pub fn construct_with_op(len: usize, op: O, mut f: impl FnMut(usize) -> O::Value) -> Self {
106        assert!(len > 0);
107        Self::with_op((0..len).map(|i| f(i)).collect(), op)
108    }
109
110    pub fn len(&self) -> usize {
111        self.data.len()
112    }
113
114    /// 累積配列の`r`番目の要素を返す(区間`[0, r)`の区間積を返す)．
115    pub fn get(&self, r: usize) -> &O::Value {
116        &self.data[r]
117    }
118
119    /// `[l, r)`の区間積を返す．
120    pub fn fold(&self, range: impl RangeBounds<usize>) -> O::Value
121    where
122        O: Group,
123    {
124        let Range { start: l, end: r } = to_open_range(range, self.data.len() - 1);
125        assert!(l <= r);
126        self.op.op(&self.data[r], &self.op.inv(&self.data[l]))
127    }
128
129    pub fn iter(&self) -> impl Iterator<Item = &O::Value> {
130        self.data.iter()
131    }
132}
133
134impl<O: Monoid> From<(Vec<O::Value>, O)> for CumulativeArray<O> {
135    fn from((v, op): (Vec<O::Value>, O)) -> Self {
136        CumulativeArray::with_op(v, op)
137    }
138}

140impl<O: Monoid, const N: usize> From<([O::Value; N], O)> for CumulativeArray<O> {
141    fn from((v, op): ([O::Value; N], O)) -> Self {
142        CumulativeArray::with_op(v.into_iter().collect(), op)
143    }
144}

232    #[test]
233    fn test_cumulative_sum() {
234        let v = vec![1, 2, 3, 4, 5];
235        let testcases = vec![
236            ((0, 5), 15),
237            ((0, 1), 1),
238            ((1, 3), 5),
239            ((3, 5), 9),
240            ((2, 4), 7),
241            ((4, 5), 5),
242            ((0, 4), 10),
243        ];
244        let cum = CumulativeSum::<i64>::from(v);
245        assert_eq!(cum.fold(..), 15);
246        assert_eq!(cum.get(5), &15);
247        for ((l, r), expected) in testcases {
248            assert_eq!(cum.fold(l..r), expected);
249        }
250
251        let cum = CumulativeSum::construct(5, |i| i as i64 + 1);
252        let testcases = vec![
253            ((0, 5), 15),
254            ((0, 1), 1),
255            ((1, 3), 5),
256            ((3, 5), 9),
257            ((2, 4), 7),
258            ((4, 5), 5),
259            ((0, 4), 10),
260        ];
261        for ((l, r), expected) in testcases {
262            assert_eq!(cum.fold(l..r), expected);
263        }
264    }

277    fn test_sum_random() {
278        let mut rng = StdRng::seed_from_u64(30);
279
280        for _ in 0..100 {
281            let v = (0..100)
282                .map(|_| rng.gen_range(-1000000000..=1000000000))

284            let cum = CumulativeSum::from(v.clone());
285            for l in 0..v.len() {
286                for r in l..=v.len() {
287                    let naive = (l..r).map(|i| v[i]).sum::<i64>();
288                    assert_eq!(cum.fold(l..r), naive);
289                }
290            }
291        }
