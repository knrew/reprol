8/// アフィン変換パラメータ: f(x) = a*x + b
9#[derive(Clone, Copy, PartialEq, Eq, Debug)]
10pub struct ActAffineElement<T> {
11    pub a: T,
12    pub b: T,
13}
14
15/// LazySegmentTreeにおけるアフィン変換作用
16///
17/// アフィン変換 + 区間和取得を行うためのAction
18#[derive(Default, Clone)]
19pub struct ActAffine<T>(PhantomData<T>);
20
21impl<T: Copy + ActAffineUtils> Monoid for ActAffine<T> {
22    type Element = ActAffineElement<T>;
23
24    #[inline]
25    fn id(&self) -> Self::Element {
26        ActAffineElement {
27            a: T::ONE,
28            b: T::ZERO,
29        }
30    }
31
32    #[inline]
33    fn op(&self, g: &Self::Element, f: &Self::Element) -> Self::Element {
34        // アフィン変換の合成:
35        // g∘f(x) = g(f(x)) = g.a*(f.a*x + f.b) + g.b = (g.a*f.a)*x + (g.a*f.b + g.b)
36        ActAffineElement {
37            a: g.a.mul_(f.a),
38            b: g.a.mul_(f.b).add_(g.b),
39        }
40    }
41}
42
43impl<T, O> Action<O> for ActAffine<T>
44where
45    O: Monoid<Element = OpAddWithLenElement<T>>,
46    T: Copy + ActAffineUtils,
47{
48    #[inline]
49    fn act(&self, f: &Self::Element, x: &O::Element) -> O::Element {
50        // 区間和に対するアフィン変換: sum' = a*sum + b*len
51        OpAddWithLenElement {
52            value: f.a.mul_(x.value).add_(f.b.mul_(x.len)),
53            len: x.len,
54        }
55    }
56}
57
58trait ActAffineUtils {
59    const ZERO: Self;
60    const ONE: Self;
61    fn add_(self, rhs: Self) -> Self;
62    fn mul_(self, rhs: Self) -> Self;
63}
64
65macro_rules! impl_actaffineutils_signed {
66    ($ty: ty) => {
67        impl ActAffineUtils for $ty {
68            const ZERO: Self = 0;
69            const ONE: Self = 1;
70
71            #[inline(always)]
72            fn add_(self, rhs: Self) -> Self {
73                self + rhs
74            }
75
76            #[inline(always)]
77            fn mul_(self, rhs: Self) -> Self {
78                self * rhs
79            }
80        }
81    };
82}
83
84macro_rules! impl_actaffineutils_unsigned {
85    ($ty: ty) => {
86        impl ActAffineUtils for $ty {
87            const ZERO: Self = 0;
88            const ONE: Self = 1;
89
90            #[inline(always)]
91            fn add_(self, rhs: Self) -> Self {
92                self.wrapping_add(rhs)
93            }
94
95            #[inline(always)]
96            fn mul_(self, rhs: Self) -> Self {
97                self.wrapping_mul(rhs)
98            }
99        }
100    };
101}
102
103macro_rules! impl_actaffineutils_for {
104    (unsigned: [$($u:ty),* $(,)?], signed: [$($s:ty),* $(,)?]$(,)?) => {
105        $( impl_actaffineutils_unsigned!($u); )*
106        $( impl_actaffineutils_signed!($s); )*
107    };
108}

135    #[test]
136    fn test_actaffine_monoid() {
137        let act = ActAffine::<i64>::default();
138
139        // 単位元: f(x) = 1*x + 0 = x
140        let id = act.id();
141        assert_eq!(id.a, 1);
142        assert_eq!(id.b, 0);
143
144        // 作用の合成
145        // f(x) = 2x + 3, g(x) = 3x + 1
146        // g∘f(x) = 3(2x + 3) + 1 = 6x + 10
147        let f = ActAffineElement { a: 2, b: 3 };
148        let g = ActAffineElement { a: 3, b: 1 };
149        let composed = act.op(&g, &f);
150        assert_eq!(composed.a, 6);
151        assert_eq!(composed.b, 10);
152
153        // 単位元との演算
154        assert_eq!(act.op(&id, &f), f);
155        assert_eq!(act.op(&f, &id), f);
156    }

164        let affine = ActAffineElement { a: 2, b: 3 };
