1//! 代入作用
2//!
3//! 各要素を定数で上書きする作用を表すモノイド．
4//! 作用の合成では，後の作用が優先される(`op(g, f)` で f を先に適用し g が上書き)．
5//! 恒等作用は`None`．
6//!
7//! [`OpMax`](crate::ops::op_max::OpMax)や[`OpMin`](crate::ops::op_min::OpMin)など，
8//! 各要素を独立に置換できるモノイドと組み合わせて使用する．
9//!
10//! # Examples
11//!
12//! ```
13//! use reprol::ops::monoid::Monoid;
14//! use reprol::ops::action::Action;
15//! use reprol::ops::act_set::ActSet;
16//! use reprol::ops::op_max::OpMax;
17//!
18//! let act = ActSet::<i64>::default();
19//! // 作用の合成: 後が優先
20//! assert_eq!(act.op(&Some(7), &Some(3)), Some(7));
21//! assert_eq!(act.op(&None, &Some(3)), Some(3));
22//! // 恒等作用
23//! assert_eq!(act.op(&act.id(), &Some(5)), Some(5));
24//! // OpMaxへの作用: 値を上書き
25//! let result = Action::<OpMax<i64>>::act(&act, &Some(10), &3);
26//! assert_eq!(result, 10);
27//! // None(恒等作用)では元の値を保持
28//! let result = Action::<OpMax<i64>>::act(&act, &None, &3);
29//! assert_eq!(result, 3);
30//! ```
31//!
32//! # Notes
33//!
34//! - [`OpAdd`](crate::ops::op_add::OpAdd)や[`OpRangeSum`](crate::ops::op_range_sum::OpRangeSum)には使用できない．
35//!   区間和と組み合わせる場合は[`ActRangeSet`](crate::ops::act_range_set::ActRangeSet)を使用する．
36
37use std::marker::PhantomData;
38
39use crate::ops::{action::Action, monoid::Monoid};

41/// 代入作用
42///
43/// 各要素を定数で上書きする作用を表すモノイド．
44/// 作用素の型は`Option<T>`で，`None`が恒等作用，`Some(v)`が値`v`への上書きを表す．
45///
46/// # Notes
47///
48/// - [`OpMax`](crate::ops::op_max::OpMax)や[`OpMin`](crate::ops::op_min::OpMin)と組み合わせて使用する．
49/// - [`OpAdd`](crate::ops::op_add::OpAdd)や[`OpRangeSum`](crate::ops::op_range_sum::OpRangeSum)には使用できない．
50#[derive(Default, Clone, Copy)]
51pub struct ActSet<T>(PhantomData<T>);
52
53impl<T: Clone> Monoid for ActSet<T> {
54    type Element = Option<T>;
55
56    #[inline]
57    fn op(&self, g: &Self::Element, f: &Self::Element) -> Self::Element {
58        if g.is_none() { f } else { g }.clone()
59    }
60
61    #[inline]
62    fn id(&self) -> Self::Element {
63        None
64    }
65}
66
67impl<O: Monoid> Action<O> for ActSet<O::Element>
68where
69    O::Element: Clone,
70{
71    #[inline]
72    fn act(&self, f: &Self::Element, x: &<O as Monoid>::Element) -> <O as Monoid>::Element {
73        if let Some(f) = f { f } else { x }.clone()
74    }
75}

98    #[test]
99    fn test_op_identity() {
100        let m = ActSet::<i64>::default();
101        let id = m.id();
102        let cases: &[Option<i64>] = &[None, Some(0), Some(1030), Some(-1), Some(i64::MAX)];
103        for f in cases {
104            assert_eq!(m.op(&id, f), *f, "op(id, {f:?}) = {f:?}");
105            assert_eq!(m.op(f, &id), *f, "op({f:?}, id) = {f:?}");
106        }
107    }
108
109    #[test]
110    fn test_op_associativity() {
111        let m = ActSet::<i64>::default();
112        let vals: &[Option<i64>] = &[None, Some(3), Some(1030), Some(-7)];
113        for h in vals {
114            for g in vals {
115                for f in vals {
116                    let lhs = m.op(&m.op(h, g), f);
117                    let rhs = m.op(h, &m.op(g, f));
118                    assert_eq!(
119                        lhs, rhs,
120                        "op(op({h:?}, {g:?}), {f:?}) = op({h:?}, op({g:?}, {f:?}))"
121                    );
122                }
123            }
124        }
125    }

158    fn test_act_composition_op_max() {
159        let act = ActSet::<i64>::default();
160        let vals: &[Option<i64>] = &[None, Some(3), Some(1030), Some(-7)];
161        for g in vals {
162            for f in vals {
163                let composed = act.op(g, f);
164                for &x in &[0i64, 1030, -1, i64::MAX] {
165                    let lhs = Action::<OpMax<i64>>::act(&act, &composed, &x);
166                    let rhs =
167                        Action::<OpMax<i64>>::act(&act, g, &Action::<OpMax<i64>>::act(&act, f, &x));
168                    assert_eq!(
169                        lhs, rhs,
170                        "act(op({g:?}, {f:?}), {x}) = act({g:?}, act({f:?}, {x}))"
171                    );
172                }
173            }
174        }

180    fn test_op_smoke_all_integer_types() {
181        macro_rules! test {
182            ($ty: ty) => {
183                assert_monoid::<ActSet<$ty>>();
184                let m = ActSet::<$ty>::default();
185                assert_eq!(m.id(), None, "id for {}", stringify!($ty));

192                assert_eq!(
193                    m.op(&None, &Some(3 as $ty)),
194                    Some(3 as $ty),
195                    "op(None, Some(3)) for {}",
196                    stringify!($ty)
197                );
198            };
199        }

215    #[test]
216    fn test_op_clone_only_type() {
217        let m = ActSet::<String>::default();
218
219        // 恒等元
220        assert_eq!(m.id(), None, "id() == None");
221
222        // op(Some, Some): g優先
223        assert_eq!(
224            m.op(&Some("b".to_string()), &Some("a".to_string())),
225            Some("b".to_string()),
226            "op(Some(b), Some(a)) = Some(b)"
227        );
228
229        // op(None, Some): fを返す
230        assert_eq!(
231            m.op(&None, &Some("a".to_string())),
232            Some("a".to_string()),
233            "op(None, Some(a)) = Some(a)"
234        );
235
236        // 結合則(3要素)
237        let h = Some("h".to_string());
238        let g = Some("g".to_string());
239        let f = Some("f".to_string());
240        let lhs = m.op(&m.op(&h, &g), &f);
241        let rhs = m.op(&h, &m.op(&g, &f));
242        assert_eq!(lhs, rhs, "associativity with String");
243    }
