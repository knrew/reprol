8/// LazySegmentTree用
9/// 値の区間加算を行う作用
10/// `seg.act(l..r, &x)`のように書くと[l, r)の区間にそれぞれxを加算する
11#[derive(Default, Clone)]
12pub struct ActAdd<T> {
13    _phantom: PhantomData<T>,
14}
15
16impl<T> Monoid for ActAdd<T>
17where
18    T: Copy + Integer,
19{
20    type Value = T;
21
22    fn identity(&self) -> Self::Value {
23        T::zero()
24    }
25
26    fn op(&self, &lhs: &Self::Value, &rhs: &Self::Value) -> Self::Value {
27        lhs.add_(rhs)
28    }
29}
30
31impl<O> Action<O> for ActAdd<O::Value>
32where
33    O: Monoid,
34    O::Value: Copy + Integer,
35{
36    fn act(&self, &f: &Self::Value, &x: &<O as Monoid>::Value) -> <O as Monoid>::Value {
37        x.add_(f)
38    }
39}
