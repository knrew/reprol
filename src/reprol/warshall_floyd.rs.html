3/// ワーシャルフロイド法によって頂点間の最小コストを計算する
4pub struct WarshallFloyd<T> {
5    /// 頂点の個数
6    n: usize,
7
8    /// cost[u][v]: u->vの最小コスト
9    costs: Vec<Vec<Option<T>>>,
10
11    // previous: Vec<Vec<Option<usize>>>,
12    /// 負の閉路が存在するか
13    has_negative_cycle: bool,
14}
15
16impl<T> WarshallFloyd<T>
17where
18    T: Clone + PartialOrd + Add<Output = T>,
19{
20    /// 隣接リスト表現のグラフから最短経路を計算する
21    pub fn new(g: &[Vec<(usize, T)>], zero: T) -> Self {
22        let n = g.len();
23        let mut costs = vec![vec![None; n]; n];
24        // let mut previous = vec![vec![None; n]; n];
25
26        for i in 0..n {
27            costs[i][i] = Some(zero.clone());
28        }
29
30        for i in 0..n {
31            for (j, c) in &g[i] {
32                match &costs[i][*j] {
33                    Some(x) if x <= c => {}
34                    _ => {
35                        costs[i][*j] = Some(c.clone());
36                        // previous[u][*v] = Some(u);
37                    }
38                }
39            }
40        }
41
42        for k in 0..n {
43            for i in 0..n {
44                for j in 0..n {
45                    // i->k, k->jがともに存在すれば，i->jが存在する
46                    if let Some(cost_ik) = &costs[i][k] {
47                        if let Some(cost_kj) = &costs[k][j] {
48                            let new_cost = cost_ik.clone() + cost_kj.clone();
49                            match &costs[i][j] {
50                                Some(cost_ij) if cost_ij <= &new_cost => {}
51                                _ => {
52                                    costs[i][j] = Some(new_cost);
53                                    // previous[i][j] = previous[k][j];
54                                }
55                            }
56                        }
57                    }
58                }
59            }
60        }
61
62        let has_negative_cycle = (0..n).any(|v| costs[v][v].as_ref().unwrap() < &zero);
63
64        Self {
65            n,
66            costs,
67            // previous,
68            has_negative_cycle,
69        }
70    }
71
72    pub fn len(&self) -> usize {
73        self.n
74    }
75
76    /// u->vの最小コスト
77    pub fn cost(&self, u: usize, v: usize) -> Option<&T> {
78        self.costs[u][v].as_ref()
79    }

109#[cfg(test)]
110mod tests {
111    use super::WarshallFloyd;
112
113    #[test]
114    fn test_warshall_floyd() {
115        // case 1: 連結
116        {
117            let n = 3;
118            // (u, v, c): u->vのコストがc
119            let edges: Vec<(usize, usize, i32)> = vec![(0, 1, 1), (1, 0, 2), (1, 2, 3), (2, 0, 4)];
120            let g = {
121                let mut g = vec![vec![]; n];
122                for &(u, v, c) in &edges {
123                    g[u].push((v, c));
124                }
125                g
126            };
127            let wf = WarshallFloyd::new(&g, 0);
128            assert!(!wf.has_negative_cycle());
129            assert_eq!(wf.cost(0, 0), Some(&0));
130            assert_eq!(wf.cost(0, 1), Some(&1));
131            assert_eq!(wf.cost(0, 2), Some(&4));
132            assert_eq!(wf.cost(1, 0), Some(&2));
133            assert_eq!(wf.cost(1, 1), Some(&0));
134            assert_eq!(wf.cost(1, 2), Some(&3));
135            assert_eq!(wf.cost(2, 0), Some(&4));
136            assert_eq!(wf.cost(2, 1), Some(&5));
137            assert_eq!(wf.cost(2, 2), Some(&0));
138        }
139
140        // case 2
141        {
142            let n = 5;
143            let edges: Vec<(usize, usize, i64)> =
144                vec![(0, 1, 4), (1, 2, 5), (0, 2, -10), (3, 4, 8), (4, 3, 12)];
145            let g = {
146                let mut g = vec![vec![]; n];
147                for &(u, v, c) in &edges {
148                    g[u].push((v, c));
149                }
150                g
151            };
152            let wf = WarshallFloyd::new(&g, 0);
153            assert!(!wf.has_negative_cycle());
154            assert_eq!(wf.cost(0, 0), Some(&0));
155            assert_eq!(wf.cost(0, 1), Some(&4));
156            assert_eq!(wf.cost(0, 2), Some(&-10));
157            assert_eq!(wf.cost(0, 3), None);
158            assert_eq!(wf.cost(0, 4), None);
159            assert_eq!(wf.cost(1, 0), None);
160            assert_eq!(wf.cost(1, 1), Some(&0));
161            assert_eq!(wf.cost(1, 2), Some(&5));
162            assert_eq!(wf.cost(3, 2), None);
163            assert_eq!(wf.cost(3, 4), Some(&8));
164            assert_eq!(wf.cost(4, 3), Some(&12));
165        }
166
167        // case 3: 負の閉路あり
168        {
169            let n = 4;
170            let edges: Vec<(usize, usize, i64)> = vec![
171                (0, 1, 1),
172                (0, 2, 5),
173                (1, 2, 2),
174                (1, 3, 4),
175                (2, 3, 1),
176                (3, 2, -7),
177            ];
178            let g = {
179                let mut g = vec![vec![]; n];
180                for &(u, v, c) in &edges {
181                    g[u].push((v, c));
182                }
183                g
184            };
185            let wf = WarshallFloyd::new(&g, 0);
186            assert!(wf.has_negative_cycle());
187        }
188    }
