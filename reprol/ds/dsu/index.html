Disjoint Set Union(Union-Find)

§使用例

use reprol::ds::dsu::Dsu;

let mut dsu = Dsu::new(5);

dsu.merge(0, 1);
dsu.merge(3, 4);
assert!(dsu.connected(0, 1));
assert!(!dsu.connected(0, 3));
assert_eq!(dsu.size(0), 2);
assert_eq!(dsu.count_components(), 3);
